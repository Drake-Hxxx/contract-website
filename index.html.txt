<script>
    // Signature canvas
    const canvas = document.getElementById('canvas');
    const ctx = canvas.getContext('2d');
    let drawing = false;

    canvas.addEventListener('mousedown', e => {
        drawing = true;
        ctx.moveTo(e.offsetX, e.offsetY);
    });

    canvas.addEventListener('mousemove', e => {
        if (drawing) {
            ctx.lineTo(e.offsetX, e.offsetY);
            ctx.stroke();
        }
    });

    canvas.addEventListener('mouseup', () => {
        drawing = false;
    });

    function clearCanvas() {
        ctx.clearRect(0, 0, canvas.width, canvas.height);
    }

    function downloadContract() {
        const name = document.getElementById('fullname').value;
        const email = document.getElementById('email').value;
        const date = document.getElementById('date').value;

        if (!name || !email || !date) {
            alert('Please fill in all the fields and sign.');
            return;
        }

        const contractText = `
Publicis Groupe Remote Work Employment Agreement

This Employment Agreement ("Agreement") is made and entered into on ${date}, by and between:

Employer: Publicis Groupe ("Company")
Employee: ${name} (${email})

The Employee agrees to the terms outlined in this Remote Work Employment Agreement, including:
- A 5-day probationary period
- Remote work expectations
- Confidentiality requirements
- Eligibility for benefits upon successful completion

Signed on: ${date}

Employee Signature:
`;

        const signatureImage = canvas.toDataURL('image/png');

        const pdf = new window.jspdf.jsPDF();
        pdf.setFontSize(12);
        pdf.text(contractText, 10, 20);
        pdf.addImage(signatureImage, 'PNG', 10, 100, 180, 60);
        pdf.text(`Employee Name: ${name}`, 10, 170);
        pdf.text(`Date: ${date}`, 10, 180);

        pdf.save(`Remote_Work_Agreement_${name}.pdf`);
    }
</script>
